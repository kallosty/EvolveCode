using UnityEngine;

public class Player : MonoBehaviour
{
    public static Player Instance;
    public Animator PlayerAnimator;

    public PlayerMotion.AnimationMotion PlayerCurrentMotion;
    public PlayerWeapon.PlayerWeaponState PlayerCurrentWeapon;
    public float PlayerDirectionValue;

    private PlayerController _playerController;
    public PlayerInteraction _playerInteraction {get; private set;}

    private void Awake()
    {
        Instance = this;
        PlayerAnimator = GetComponent();
        _playerController = GetComponentInChildren();

        PlayerCurrentMotion = GetComponentInChildren().CurrentMotion;
        PlayerCurrentWeapon = GetComponentInChildren().CurrentPlayerWeaponState;
        PlayerDirectionValue = GetComponentInChildren().IndicatorRotation;

        _playerInteraction = GetComponent();
    }

    void Start()
    {
        _playerController?.Initialize();
    }

    void Update()
    {
        PlayerCurrentWeapon = GetComponentInChildren().CurrentPlayerWeaponState;
        _playerController?.HandleUpdate();
    }
}
